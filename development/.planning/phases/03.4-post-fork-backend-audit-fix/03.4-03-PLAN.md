```python
"""
Restrict original ERPNext Accounts reports so only System Manager can see them.
University users see university_finance versions instead (Fee Receivable, Vendor Payable, etc.)

Usage:
    bench --site university.local execute university_erp.scripts.restrict_erpnext_reports.run
"""
import frappe

# ERPNext Accounts reports that have university_finance equivalents
DUPLICATE_REPORTS = [
    "Accounts Receivable",
    "Accounts Receivable Summary",
    "Accounts Payable",
    "Accounts Payable Summary",
    "General Ledger",
    "Trial Balance",
    "Balance Sheet",
    "Profit and Loss Statement",
    "Cash Flow",
    "Consolidated Financial Statement",
    "Customer Ledger Summary",
    "Supplier Ledger Summary",
    "Bank Reconciliation Statement",
    "Bank Clearance Summary",
    "Payment Period Based On Invoice Date",
    "Sales Register",
    "Purchase Register",
    "Item-wise Sales Register",
    "Item-wise Purchase Register",
    "Gross Profit",
    "Budget Variance Report",
]

def run():
    frappe.flags.ignore_permissions = True
    restricted = 0
    skipped = 0

    for report_name in DUPLICATE_REPORTS:
        if not frappe.db.exists("Report", report_name):
            print(f"  SKIP (not found): {report_name}")
            skipped += 1
            continue

        report = frappe.get_doc("Report", report_name)

        # Clear existing roles
        report.roles = []

        # Add System Manager only
        report.append("roles", {
            "role": "System Manager"
        })

        report.save(ignore_permissions=True)
        restricted += 1
        print(f"  RESTRICTED: {report_name} -> System Manager only")

    frappe.db.commit()
    print(f"\nDone: {restricted} reports restricted, {skipped} skipped (not found)")
```
